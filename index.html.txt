// ---- 农历工具 ----
// 返回指定公历年农历正月初一的 Date (本地)
function lunarNewYearDay(y){
  // 用已知锚点：1900年正月一日 ≈ 1900-01-31（查表确认）
  // 改用积日法：从 1900 正月初一为基准逐月累加
  let daySum=0;
  for(let i=1900;i<y;i++) daySum+=lunarYearDays(i);
  // 1900年正月初一 = 1900-01-31
  let base=new Date(1900,0,31);
  base.setDate(base.getDate()+daySum);
  return base;
}

function lunarYearDays(y){
  let sum=0; let info=lunarInfo[y-1900];
  for(let i=0x8000;i>=0x8;i>>=1) sum+= (info&i)?30:29;
  if(leapMonth(y)) sum+= leapDays(y);
  return sum;
}
function leapMonth(y){return lunarInfo[y-1900]&0xf}
function leapDays(y){
  if(leapMonth(y)) return (lunarInfo[y-1900]&0x10000)?30:29;
  return 0;
}
function monthDays(y,m){ // m: 1=正月..12=腊月
  return (lunarInfo[y-1900]&(0x10000>>m))?30:29;
}

// 公历→农历
function solarToLunar(sy,sm,sd){
  // 基准：1900-01-31 = 1900年正月初一
  let base=new Date(1900,0,31);
  let obj=new Date(sy,sm-1,sd);
  let offset=Math.round((obj-base)/86400000); // 距1900正月初一的天数
  let y=1900,leap=false;
  for(;y<2101;y++){
    let d=lunarYearDays(y);
    if(offset<d) break;
    offset-=d; y++; // 修正
  }
  // 重新精确定位
  offset=Math.round((obj-base)/86400000);
  y=1900;
  for(;y<2101;y++){
    let d=lunarYearDays(y);
    if(offset<d) break;
    offset-=d;
  }
  // 此时 offset ∈ [0, lunarYearDays(y))
  let isLeap=false;
  let lm=leapMonth(y);
  let left=offset;
  let mon;
  for(mon=1;mon<=12;mon++){
    if(lm===mon && left===monthDays(y,mon)){isLeap=true;mon++;break} // 跨过闰月边界
    // 简化：逐个减
  }
  // 重来更清晰地
  return _findLunar(y,offset);
}

function _findLunar(y,off){
  let isLeap=false;
  let lm=leapMonth(y);
  let cur=off;
  let m;
  for(m=1;m<=12;m++){
    if(lm===m && !isLeap){
      if(cur<leapDays(y)){isLeap=true;break;}
      cur-=leapDays(y);
      continue;
    }
    let md=monthDays(y,m);
    if(cur<md){break;}
    cur-=md;
  }
  if(m>12){m=12;cur=0} // safety
  return {year:y,month:m,isLeap:isLeap,day:cur+1};
}

// 年柱（以立春为界 → 简化：1月4日前算前一年）
function yearGZ(y,m){
  let yy=y; if(m<=2) yy--;  // 粗略以2月为界（够用，阴盘不苛求年柱换月点）
  let idx=(yy-4)%60; if(idx<0)idx+=60;
  return idx; // 甲子序号 0~59
}
function yearGZStr(y,m){
  let idx=yearGZ(y,m);
  return gz(idx);
}

// 日柱（锚点法，1900-01-01 = 甲戌 = 序号10）
function dayGZIdx(y,m,d){
  let t=Date.UTC(y,m-1,d);
  let base=Date.UTC(1900,0,1); // 1900-01-01
  // 1900-01-01 甲戌: 天干甲=0,戌=10 → 序号满足 x%10=0,x%12=10 → x=10
  let diff=Math.round((t-base)/86400000);
  let r=((10+diff)%60+60)%60;
  return r;
}
function dayGZStr(y,m,d){return gz(dayGZIdx(y,m,d))}
